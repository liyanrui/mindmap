\usemodule[mindmap]
\startMPpage
% branch length increases by a factor of two.
mind.stretch(2);

mind.enter("This is $A$", 30);
    mind("This is $A_1$", 45);
    mind("This is $A_2$", 15);
mind.exit;

mind.enter("This is $B$", -20);
    mind("This is $B_1$", -5);
    mind("This is $B_2$", -35);
mind.exit;
\stopMPpage
