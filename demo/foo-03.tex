    mind("This is $A_2$", 15);

        mind("This is $B_{1,1}$", 25);
        mind("This is $B_{1,2}$", -5);
        mind("This is $B_{1,3}$", -35);
    mind.exit;
    mind("This is $B_2$", -35);
mind.exit;
\stopMPpage
